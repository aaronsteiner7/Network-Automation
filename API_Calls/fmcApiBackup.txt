#Request an API auth token

curl -k -v --request POST "https://<fmc ip address>/api/fmc_platform/v1/auth/generatetoken" \
-H "Authorization: Basic <NODE ID>" 2>&1 | grep -i "X-auth-access-token" | awk '{print $3}'


#API call to pull backup

curl -kX 'GET' \
  'https://<fmc ip address>/api/fmc_config/v1/domain/<domain_id>/backup/downloadbackup/manager' \
  -H 'accept: application/json' \
  -H 'X-auth-access-token: <token printed from above>' \
  >/backup/path/location/fmc-backup.tar
